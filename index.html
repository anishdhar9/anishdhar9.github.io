Parameters swept

Output

# Predictor: forward difference
U_pred[:, 1:-1] = U[:, 1:-1] - dt/dx * (F[:, 2:] - F[:, 1:-1])

# Corrector: backward difference on predicted fluxes
F_pred = compute_flux(U_pred)
